```javascript
beforeCreate() { /*生命周期初始化 绑定事件*/
      alert('beforeCreate');
    },
    created() { /*可以拿到数据 */
      alert('create');
    },
    beforeMount() { /*渲染之前*/
      alert('beforeMount');
    },
    mounted() { /*获取dom  发送ajax*/
      alert('mount');
    },
    beforeUpdate() { /*更新前*/
      alert('beforeUpdate');
    },
    updated() { /*更新后*/
      alert('update');
    },
    beforeDestroy() { /*销毁前  清除定时器*/
      alert('beforeDestroy');
    },
    destroyed() { /*销毁后*/
      alert('destroy');
    },
```
